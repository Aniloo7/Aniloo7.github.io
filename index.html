interface parrot{ public void eat(); public void fly(); } interface cuckoo{ public void humming(); public void eat(); } class owl implements parrot,cuckoo{ public void eat(){ System.out.println("eating"); } public void humming(){ System.out.println("humming"); } public void fly(){ System.out.println("flying "); } } class TestMultipleInheritanceUsingInterfaces{ public static void main(String args[]){ owl o = new owl(); o.eat(); o.humming(); o.fly(); } }
